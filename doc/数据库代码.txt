/*
 * 数据库初始化
 */
CREATE DATABASE IF NOT EXISTS camp_system DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE camp_system;

-- ========================================================
-- 模块一：技术辅助表 (供 Java 调用)
-- ========================================================

-- 序列表：Java 后端会通过操作这张表来获取当前的自增序号
CREATE TABLE IF NOT EXISTS t_user_seq (
    year INT PRIMARY KEY COMMENT '年份',
    current_seq INT DEFAULT 0 COMMENT '当前该年份的最大序号'
) COMMENT = '系统序列表(用于Java后端生成ID)';

-- ========================================================
-- 模块二：用户中心
-- ========================================================

CREATE TABLE t_user (
    user_id BIGINT PRIMARY KEY COMMENT '用户ID (由Java端计算生成: YYYY+000000)',
    account VARCHAR(50) NOT NULL COMMENT '账户名',
    password VARCHAR(100) NOT NULL COMMENT '密码',
    avatar VARCHAR(255) COMMENT '头像URL',
    phone VARCHAR(20) COMMENT '手机号',
    region VARCHAR(100) COMMENT '地域',
    birthday DATE COMMENT '生日',
    profession VARCHAR(50) COMMENT '职业',
    gender TINYINT DEFAULT 0 COMMENT '性别: 0未知, 1男, 2女',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '注册时间',
    status TINYINT DEFAULT 1 COMMENT '状态: 1正常, 0冻结',

    -- 安全性优化：防止账号和手机号重复
    UNIQUE KEY uk_account (account),
    UNIQUE KEY uk_phone (phone)
) COMMENT = '用户基础表';

CREATE TABLE t_certificate (
    cert_id INT AUTO_INCREMENT PRIMARY KEY COMMENT '证书ID',
    user_id BIGINT NOT NULL COMMENT '所属用户',
    type VARCHAR(50) COMMENT '证书类型',
    number VARCHAR(50) COMMENT '证书编号',
    image_url VARCHAR(255) COMMENT '证书图片',
    issue_time DATETIME COMMENT '发放时间',
    FOREIGN KEY (user_id) REFERENCES t_user(user_id) ON DELETE CASCADE
) COMMENT = '荣誉证书表';

-- ========================================================
-- 模块三：组织架构与社交
-- ========================================================

CREATE TABLE t_group_chat (
    chat_id INT AUTO_INCREMENT PRIMARY KEY COMMENT '群ID',
    name VARCHAR(100) NOT NULL COMMENT '群聊名称',
    type VARCHAR(20) COMMENT '类型: 班级群/大组群/小组群',
    content TEXT COMMENT '群公告/简介',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP
) COMMENT = '群聊信息资源池';

CREATE TABLE t_camp_type (
    type_id INT AUTO_INCREMENT PRIMARY KEY COMMENT '类型ID',
    level VARCHAR(20) COMMENT '等级代码',
    level_name VARCHAR(50) COMMENT '等级名称'
) COMMENT = '营期类型定义';

CREATE TABLE t_camp (
    camp_id INT AUTO_INCREMENT PRIMARY KEY COMMENT '营期ID',
    type_id INT NOT NULL,
    name VARCHAR(100) NOT NULL,
    intro TEXT,
    start_time DATETIME,
    end_time DATETIME,
    status TINYINT DEFAULT 0,
    FOREIGN KEY (type_id) REFERENCES t_camp_type(type_id)
) COMMENT = '营期表';

CREATE TABLE t_class (
    class_id INT AUTO_INCREMENT PRIMARY KEY COMMENT '班级ID',
    camp_id INT NOT NULL,
    chat_id INT,
    name VARCHAR(100) NOT NULL,
    FOREIGN KEY (camp_id) REFERENCES t_camp(camp_id) ON DELETE CASCADE,
    FOREIGN KEY (chat_id) REFERENCES t_group_chat(chat_id) ON DELETE SET NULL
) COMMENT = '班级表';

CREATE TABLE t_big_group (
    big_group_id INT AUTO_INCREMENT PRIMARY KEY COMMENT '大组ID',
    class_id INT NOT NULL,
    chat_id INT,
    name VARCHAR(100) NOT NULL,
    FOREIGN KEY (class_id) REFERENCES t_class(class_id) ON DELETE CASCADE,
    FOREIGN KEY (chat_id) REFERENCES t_group_chat(chat_id) ON DELETE SET NULL
) COMMENT = '大组表';

CREATE TABLE t_small_group (
    small_group_id INT AUTO_INCREMENT PRIMARY KEY COMMENT '小组ID',
    big_group_id INT NOT NULL,
    chat_id INT,
    name VARCHAR(100) NOT NULL,
    FOREIGN KEY (big_group_id) REFERENCES t_big_group(big_group_id) ON DELETE CASCADE,
    FOREIGN KEY (chat_id) REFERENCES t_group_chat(chat_id) ON DELETE SET NULL
) COMMENT = '小组表';

-- ========================================================
-- 模块四：档案、权限与教学
-- ========================================================

CREATE TABLE t_camp_enrollment (
    enroll_id INT AUTO_INCREMENT PRIMARY KEY COMMENT '档案ID',
    user_id BIGINT NOT NULL,
    camp_id INT NOT NULL,
    class_id INT,
    big_group_id INT,
    small_group_id INT,
    is_completed TINYINT DEFAULT 0,
    UNIQUE KEY uk_user_camp (user_id, camp_id),
    FOREIGN KEY (user_id) REFERENCES t_user(user_id),
    FOREIGN KEY (camp_id) REFERENCES t_camp(camp_id)
) COMMENT = '学员档案表';

CREATE TABLE t_duty_assignment (
    assignment_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    camp_id INT,
    duty_type VARCHAR(50) NOT NULL,
    FOREIGN KEY (user_id) REFERENCES t_user(user_id)
) COMMENT = '职责任命表';

CREATE TABLE t_duty_scope (
    scope_id INT AUTO_INCREMENT PRIMARY KEY,
    assignment_id INT NOT NULL,
    target_id INT NOT NULL,
    target_type VARCHAR(20) NOT NULL,
    FOREIGN KEY (assignment_id) REFERENCES t_duty_assignment(assignment_id) ON DELETE CASCADE
) COMMENT = '职责范围表';

CREATE TABLE t_camp_plan (
    plan_id INT AUTO_INCREMENT PRIMARY KEY,
    camp_id INT NOT NULL,
    plan_date DATE NOT NULL,
    graphic_url VARCHAR(255),
    video_url VARCHAR(255),
    is_finished TINYINT DEFAULT 0,
    FOREIGN KEY (camp_id) REFERENCES t_camp(camp_id)
) COMMENT = '营期计划表';

CREATE TABLE t_homework (
    homework_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    plan_id INT NOT NULL,
    content TEXT,
    submit_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY uk_user_plan (user_id, plan_id),
    FOREIGN KEY (user_id) REFERENCES t_user(user_id),
    FOREIGN KEY (plan_id) REFERENCES t_camp_plan(plan_id)
) COMMENT = '作业表';

CREATE TABLE t_rules (
    rule_id INT AUTO_INCREMENT PRIMARY KEY,
    camp_id INT NOT NULL,
    rule_type VARCHAR(50),
    config_json JSON,
    FOREIGN KEY (camp_id) REFERENCES t_camp(camp_id)
) COMMENT = '规则配置表';
